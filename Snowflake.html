float x = -45; float y = 325; float dx = 25; float angle = 0;
snowflake[] snow;

void setup(){
 size(600,600); background(255);
snow = new snowflake[30];
for(int i=0; i<30; i++){
snow[i] = new snowflake(x,y+sin(angle)*300,80,color(255,255,255,200));
angle = angle + 1; x = x + dx;
}
for (int i = 0; i < 30; i++) {
 for (int j = 0; j < 30; j++){
   stroke(200,200,200,40);
   line(snow[i].x, snow[i].y, snow[j].x, snow[j].y);
 }
}

for (int i = 29; i >= 0; i--) {
 snow[i].center(); snow[i].levelup();
}
}

class snowflake{
 float x; float y; float rad; color c;
snowflake(float tempX, float tempY, float tempRad, color tempC){
x = tempX; y = tempY; rad = tempRad; c = tempC;
}

void center(){
 noStroke(); fill(c);
ellipse(x,y,rad,rad);
fill(255,100,100,180);
ellipse(x,y,6,6);
}

void levelup(){
 for(int i = 0 ; i<6 ; i ++){
 fill(187,222,251,100);
 float angle = PI/2 + i * PI/3;
 ellipse(cos(angle)*50+x,sin(angle)*50+y,45,45);
level2(cos(angle)*50+x,sin(angle)*50+y);
}
}

void level2(float q, float w){
 for(int i = 0 ; i<6 ; i ++){
 fill(187,222,251,100);
 float angle = PI/2 + i * PI/3;
 ellipse(cos(angle)*20+q,sin(angle)*20+w,20,20);
 level3(cos(angle)*20+q,sin(angle)*20+w);
}
}

void level3(float q, float w){
 for(int i = 0 ; i<6 ; i ++){
 fill(187,222,251,100);
 float angle = PI/2 + i * PI/3;
 ellipse(cos(angle)*10+q,sin(angle)*10+w,5,5);
}
}
}
